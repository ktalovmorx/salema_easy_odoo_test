	def unitario(cls):
		instancias = {}
		def get_instance(*args, **kwargs):
			if cls not in instancias:
				instancias[cls] = cls(*args, **kwargs)
			return instancias[cls]
		return get_instance

	@unitario
	class ElSolitario(object):
		def __init__(self, name):
			self.name = name

	soli1 = ElSolitario("Solitario 1")
	soli2 = ElSolitario("Solitario 2")
